41 void SchedulingASAP::addNewInstruction(Instruction &I)
42 {
43  Instr i=identifyInstr(I);
44 
45  errs()<<"Fetched: " << I << "; recognized: " << i << "\n";
46 
47  switch(i)
48  {
49  case alloc:
50  {
51  //Number of allocated data (e.g. INT=1, VECTOR=N)
52  int arraySize=1;
53 
54  errs() << "Alloca detected!\n";
55  AllocaInst *a=dyn_cast<AllocaInst>(&I);
56  Type *ty=(a->getAllocatedType());
57 
58  if(ty->isArrayTy())
59  {
60  errs()<< "\tArray detected!\n";
61  arraySize= (int) ty->getArrayNumElements();
62  errs()<< "\tDimension: "<< arraySize << "\n";
63  }
64 
65  IT.AddAllocaInstructionToList(Timer, (int *) &I, arraySize);
66  }
67  break;
68 
69  case load:
70  {
71  errs() << "Load detected!\n";
72 
73  int * parentReg;
74  int index=0; //Default value for traditional load instructions (no array)
75 
76  //Check if the parsed instruction refers to an array
77  if(infoAboutPtr.valid==true)
78  {
79  //Check if the assumption is correct
80  if(infoAboutPtr.ptrName==(int *) I.getOperand(0))
81  {
82  errs() << "\tCheck on the index is correct!\n";
83  //The element derives from an array
84  parentReg=infoAboutPtr.srcReg;
85  index=infoAboutPtr.index;
86  infoAboutPtr.valid=false;
87  } else {
88  llvm_unreachable("Error in SchedulingASAP: load instruction after GEP refers to an unknown pointer.");
89  }
90 
91  } else {
92  parentReg = (int *) I.getOperand(0);
93  }
94 
95  if (!IT.isParentValid(parentReg,index)){
96  parentReg = getRealParent(parentReg);
97  }
98 
99  IT.AddInstructionToList(Timer, Timer, I.getOpcodeName(), (int *) &I, parentReg, nullptr);
100  aliasMap.insert(std::pair<int * const, int * const>((int *)&I, parentReg)); //Check the order
101  }
102  break;
103 
104  case store:
105  {
106  errs() << "Store detected!\n";
107 
108  int * destReg;
109  int index=0; //Default value for traditional load instructions (no array)
110 
111  //Check if the parsed instruction refers to an array
112  if(infoAboutPtr.valid==true)
113  {
114  //Check if the assumption is correct
115  if(infoAboutPtr.ptrName==(int *) I.getOperand(0))
116  {
117  //The element derives from an array
118  destReg=infoAboutPtr.srcReg;
119  index=infoAboutPtr.index;
120  infoAboutPtr.valid=false;
121  } else {
122  llvm_unreachable("Error in SchedulingASAP: store instruction after GEP refers to an unknown pointer.");
123  }
124 
125  } else {
126  destReg=(int *)I.getOperand(1);
127  }
128 
129 
130  //Invalidate the data previously declared through 'alloca' statement
131  IT.invalidateParent(destReg,index);
132 
133  //Update the alias map!!!!
134 
135  //Set the relation between the newest data and the invalid one
136  aliasMap.insert(std::pair<int * const, int * const>((int *)I.getOperand(1), (int *)I.getOperand(0))); //Check the order
137  }
138  break;
139 
140  case binary:
141  {
142  errs() << "Binary detected!\n";
143 
144  //Check if the input function is a NOT
145  if(isThisNot(I)){
146  errs() << "\tNOT instruction detected!\n";
147  changeParentInNOT((int *)I.getOperand(0),I.getOpcodeName(),(int *) &I);
148 
149  } else {
150 
151  //Define the minimum time in which schedule the operation
152  //(i.e. when all the operands are ready)
153  int Top1=IT.getAvailableTime((int *)I.getOperand(0));
154  int Top2=IT.getAvailableTime((int *)I.getOperand(1));
155 
156  int Tex=std::max(Top1,Top2);
157 
158  //The result will be written at the next clock cycle
159  Tex++;
160 
161  IT.AddInstructionToList(Tex, Tex, I.getOpcodeName(), (int *) &I, (int *)I.getOperand(0), (int *)I.getOperand(1));
162  }
163 
164  }
165  break;
166 
167  case ptr:
168  {
169  errs() << "Pointer detected!\n";
170 
171  int index;
172 
173  //To identify the kind of pointer: array
174  GetElementPtrInst *GEP = dyn_cast<GetElementPtrInst>(&I);
175  if (!GEP){
176  // The instruction doesn't represent a pointer to an array
177  llvm_unreachable("SchedulingASAP error: current instruction not recognized.\n");
178  }
179 
180  //Get the index
181  Value *offset = GEP->getOperand(2);
182  ConstantInt* CI = dyn_cast<ConstantInt>(offset);
183 
184  index=CI->getSExtValue();
185 
186  // Set the useful information for the following input instruction
187  infoAboutPtr.ptrName=(int *)&I;
188  infoAboutPtr.srcReg=(int *) GEP->getPointerOperand();
189  infoAboutPtr.index=index;
190  infoAboutPtr.valid=true;
191  }
192  break;
193 
194  case ret:
195  {
196  //To be implemented: actually it doesn't do anything!
197  }
198  break;
199 
200  default:
201  {
202  IT.printIT();
203  IT.printAllocData();
204  printAliasMap();
205  llvm_unreachable("SchedulingASAP error: current instruction not recognized.\n");
206  }
207 
208  }
209 }

230 SchedulingASAP::Instr SchedulingASAP::identifyInstr(Instruction &I){
231 
232  //Identification of the kind of instruction (see Instruction.def)
233  //The instruction set is limited considering that the LLVM IR has been
234  //previously optimized by the Transform Passes.
235  if(isa<AllocaInst>(I))
236  return alloc;
237 
238  if(isa <LoadInst> (I))
239  return load;
240 
241  if(isa <StoreInst> (I))
242  return store;
243 
244  if(isa <BinaryOperator> (I))
245  return binary;
246 
247  if(isa <ReturnInst> (I))
248  return ret;
249 
250  if(isa<GetElementPtrInst>(I))
251  return ptr;
252 
253  //Not valid instruction
254  return unknown;
255 }

258 bool SchedulingASAP::isThisNot(Instruction &I){
259  std::string logicPort;
260  int operand;
261 
262  logicPort=I.getOpcodeName();
263 
264  //Chech if the second operand is an integer value
265  if(ConstantInt* CI=dyn_cast<ConstantInt>(I.getOperand(1))){
266 
267  //Converting the operand into an integer value (operand)
268  if (CI->getBitWidth() <= 32) {
269  operand = CI->getSExtValue();
270 
271  if(logicPort=="xor"&&operand==-1)
272  {
273  errs() << "The logic port is a NOT!\n";
274  return true;
275  }
276  }
277  }
278 
279  //Case in which the second operand is not a -1
280  errs() << "The logic port is not a NOT!\n";
281  return false;
